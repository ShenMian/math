190 constexpr T determinant() const
191 {
192 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
193
194 const auto& mat = *this;
195 const float a0 = mat(0, 0) * mat(1, 1) - mat(0, 1) * mat(1, 0);
196 const float a1 = mat(0, 0) * mat(1, 2) - mat(0, 2) * mat(1, 0);
197 const float a2 = mat(0, 0) * mat(1, 3) - mat(0, 3) * mat(1, 0);
198 const float a3 = mat(0, 1) * mat(1, 2) - mat(0, 2) * mat(1, 1);
199 const float a4 = mat(0, 1) * mat(1, 3) - mat(0, 3) * mat(1, 1);
200 const float a5 = mat(0, 2) * mat(1, 3) - mat(0, 3) * mat(1, 2);
201 const float b0 = mat(2, 0) * mat(3, 1) - mat(2, 1) * mat(3, 0);
202 const float b1 = mat(2, 0) * mat(3, 2) - mat(2, 2) * mat(3, 0);
203 const float b2 = mat(2, 0) * mat(3, 3) - mat(2, 3) * mat(3, 0);
204 const float b3 = mat(2, 1) * mat(3, 2) - mat(2, 2) * mat(3, 1);
205 const float b4 = mat(2, 1) * mat(3, 3) - mat(2, 3) * mat(3, 1);
206 const float b5 = mat(2, 2) * mat(3, 3) - mat(2, 3) * mat(3, 2);
207 return (a0 * b5 - a1 * b4 + a2 * b3 + a3 * b2 - a4 * b1 + a5 * b0);
208 }

298 MatrixT& rotate_x(float angle)
299 {
300 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
301
302 auto& mat = *this;
303
304 const auto cos = std::cos(angle);
305 const auto sin = std::sin(angle);
306
307 const auto m01 = mat(0, 1);
308 const auto m11 = mat(1, 1);
309 const auto m21 = mat(2, 1);
310
311 mat(0, 1) = mat(0, 1) * cos + mat(0, 2) * sin;
312 mat(1, 1) = mat(1, 1) * cos + mat(1, 2) * sin;
313 mat(2, 1) = mat(2, 1) * cos + mat(2, 2) * sin;
314
315 mat(0, 2) = mat(0, 2) * cos - m01 * sin;
316 mat(1, 2) = mat(1, 2) * cos - m11 * sin;
317 mat(2, 2) = mat(2, 2) * cos - m21 * sin;
318
319 return *this;
320 }

403 void decompose(Vector3T<T>* translation, QuaternionT<T>* rotation, Vector3T<T>* scale) const
404 {
405 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
406
407 const auto& mat = *this;
408 if(translation)
409 {
410 translation->x() = mat(3, 0);
411 translation->y() = mat(3, 1);
412 translation->z() = mat(3, 2);
413 }
414
415 if(scale == nullptr && rotation == nullptr)
416 return;
417
418 auto axis_x = mat.row(0);
419 auto axis_y = mat.row(1);
420 auto axis_z = mat.row(2);
421 // auto axis_x = Vector3T<T>({mat(0, 0), mat(0, 1), mat(0, 2)});
422 // auto axis_y = Vector3T<T>({mat(1, 0), mat(1, 1), mat(1, 2)});
423 // auto axis_z = Vector3T<T>({mat(2, 0), mat(2, 1), mat(2, 2)});
424 const auto scale_x = axis_x.norm();
425 const auto scale_y = axis_y.norm();
426 auto scale_z = axis_z.norm();
427 if(determinant() < 0)
428 scale_z = -scale_z;
429
430 if(scale)
431 {
432 scale->x() = scale_x;
433 scale->y() = scale_y;
434 scale->z() = scale_z;
435 }
436
437 if(rotation)
438 {
439 if(scale_x < std::numeric_limits<T>::epsilon() || scale_y < std::numeric_limits<T>::epsilon() ||
440 scale_z < std::numeric_limits<T>::epsilon()) [[unlikely]]
441 return; // 除数过于接近 0, 无法完成计算
442
443 axis_x.normalize();
444 axis_y.normalize();
445 axis_z.normalize();
446 const float trace = axis_x.x() + axis_y.y() + axis_z.z() + 1.0f;
447
448 if(trace > std::numeric_limits<T>::epsilon())
449 {
450 const float s = 0.5f / std::sqrt(trace);
451 rotation->w = 0.25f / s;
452 rotation->x = (axis_y.z() - axis_z.y()) * s;
453 rotation->y = (axis_z.x() - axis_x.z()) * s;
454 rotation->z = (axis_x.y() - axis_y.x()) * s;
455 }
456 else
457 {
458 if(axis_x.x() > axis_y.y() && axis_x.x() > axis_z.z())
459 {
460 const float s = 0.5f / std::sqrt(1.0f + axis_x.x() - axis_y.y() - axis_z.z());
461 rotation->w = (axis_y.z() - axis_z.y()) * s;
462 rotation->x = 0.25f / s;
463 rotation->y = (axis_y.x() + axis_x.y()) * s;
464 rotation->z = (axis_z.x() + axis_x.z()) * s;
465 }
466 else if(axis_y.y() > axis_z.z())
467 {
468 const float s = 0.5f / std::sqrt(1.0f + axis_y.y() - axis_x.x() - axis_z.z());
469 rotation->w = (axis_z.x() - axis_x.z()) * s;
470 rotation->x = (axis_y.x() + axis_x.y()) * s;
471 rotation->y = 0.25f / s;
472 rotation->z = (axis_z.y() + axis_y.z()) * s;
473 }
474 else
475 {
476 const float s = 0.5f / std::sqrt(1.0f + axis_z.z() - axis_x.x() - axis_y.y());
477 rotation->w = (axis_x.y() - axis_y.x()) * s;
478 rotation->x = (axis_z.x() + axis_x.z()) * s;
479 rotation->y = (axis_z.y() + axis_y.z()) * s;
480 rotation->z = 0.25f / s;
481 }
482 }
483 }

748 static MatrixT create_rotation(const QuaternionT<T>& q)
749 {
750 const float x2 = q.x + q.x;
751 const float y2 = q.y + q.y;
752 const float z2 = q.z + q.z;
753
754 const float xx2 = q.x * x2;
755 const float yy2 = q.y * y2;
756 const float zz2 = q.z * z2;
757 const float xy2 = q.x * y2;
758 const float xz2 = q.x * z2;
759 const float yz2 = q.y * z2;
760 const float wx2 = q.w * x2;
761 const float wy2 = q.w * y2;
762 const float wz2 = q.w * z2;
763
764 MatrixT mat;
765 mat(0, 0) = 1.0f - yy2 - zz2;
766 mat(0, 1) = xy2 + wz2;
767 mat(0, 2) = xz2 - wy2;
768
769 mat(1, 0) = xy2 - wz2;
770 mat(1, 1) = 1.0f - xx2 - zz2;
771 mat(1, 2) = yz2 + wx2;
772
773 mat(2, 0) = xz2 + wy2;
774 mat(2, 1) = yz2 - wx2;
775 mat(2, 2) = 1.0f - xx2 - yy2;
776 return mat;
777 }

785 static MatrixT create_rotation(float angle, Vector3T<T> axis)
786 {
787 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
788 MATH_DEBUG_CHECK(axis == Vector3T<T>::unit_x || axis == Vector3T<T>::unit_y || axis == Vector3T<T>::unit_z);
789
790 axis.normalize();
791 const float x = axis.x;
792 const float y = axis.y;
793 const float z = axis.z;
794
795 const float sin = std::sin(angle);
796 const float cos = std::cos(angle);
797 const float t = 1.0f - cos;
798 const float tx = t * x;
799 const float ty = t * y;
800 const float tz = t * z;
801 const float txy = tx * y;
802 const float txz = tx * z;
803 const float tyz = ty * z;
804 const float sx = sin * x;
805 const float sy = sin * y;
806 const float sz = sin * z;
807
808 auto mat = MatrixT::identity();
809 mat(0, 0) = cos + tx * x;
810 mat(0, 1) = txy + sz;
811 mat(0, 2) = txz - sy;
812
813 mat(1, 0) = txy - sz;
814 mat(1, 1) = cos + ty * y;
815 mat(1, 2) = tyz + sx;
816
817 mat(2, 0) = txz + sy;
818 mat(2, 1) = tyz - sx;
819 mat(2, 2) = cos + tz * z;
820 return mat;
821 }

847 static MatrixT<T, 4, 4> perspective(T vFOV, T aspect, T near, T far)
848 {
849 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
850 static_assert(std::is_floating_point_v<T>, "element type must be floating point");
851 MATH_DEBUG_CHECK(!equal(aspect, 0.f));
852 MATH_DEBUG_CHECK(near != far);
853
854 const auto tanHalfFOV = std::tan(vFOV / 2.f);
855
856 auto mat = MatrixT::identity();
857 mat(0, 0) = T(1) / (aspect * tanHalfFOV);
858 mat(1, 1) = T(1) / tanHalfFOV;
859 mat(2, 2) = far / (far - near);
860 mat(2, 3) = T(1);
861 mat(3, 2) = -(far * near) / (far - near);
862 return mat;
863 }

892 {
893 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
894 static_assert(std::is_floating_point_v<T>, "element type must be floating point");

896

903 mat(3, 2) = -near / (far - near);
904 mat(3, 3) = T(1);

915 static MatrixT<T, 4, 4> look_at(const Vector3T<T>& eye, const Vector3T<T>& target,
916 const Vector3T<T>& up = Vector3T<T>::up)
917 {
918 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
919 static_assert(std::is_floating_point_v<T>, "element type must be floating point");
920
921 Vector3T<T> f((target - eye).normalize()); // front
922 Vector3T<T> u(up.normalized()); // up
923 Vector3T<T> r(f.cross(up).normalize()); // right
924 u = r.cross(f);
925
926 auto mat = MatrixT::identity();
927
928 mat(0, 0) = r.x();
929 mat(1, 0) = r.y();
930 mat(2, 0) = r.z();
931
932 mat(0, 1) = u.x();
933 mat(1, 1) = u.y();
934 mat(2, 1) = u.z();
935
936 mat(0, 2) = -f.x();
937 mat(1, 2) = -f.y();
938 mat(2, 2) = -f.z();
939
940 mat(3, 0) = -r.dot(eye);
941 mat(3, 1) = -u.dot(eye);
942 mat(3, 2) = f.dot(eye);
943
944 return mat;
945 }
