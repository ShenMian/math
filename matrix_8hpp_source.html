224 friend std::ostream& operator<<(std::ostream& stream, const MatrixT& mat)
225 {
226 size_t max = 0;
227 for(size_t r = 0; r < R; r++)
228 {
229 for(size_t c = 0; c < C; c++)
230 {
231 std::ostringstream stm;
232 stm << mat(r, c);
233 const auto size = stm.str().size();
234 max = std::max(max, size);
235 }
236 }
237 max = std::min<size_t>(max, 6);
238
239 for(size_t r = 0; r < R; r++)
240 {
241 for(size_t c = 0; c < C; c++)
242 {
243 std::ostringstream stm;
244 stm.setf(std::ios::fixed);
245 stm.precision(6);
246 stm << mat(r, c);
247 auto str = stm.str();
248 str.erase(str.find_last_not_of('0') + 1);
249 if(str.back() == '.')
250 str.pop_back();
251 stream.width(max);
252 stream << str << ' ';
253 }
254 stream << '\n';
255 }
256 return stream;
257 }
