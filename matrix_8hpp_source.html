175 constexpr T determinant() const
176 {
177 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
178
179 const auto& mat = *this;
180 const float a0 = mat(0, 0) * mat(1, 1) - mat(0, 1) * mat(1, 0);
181 const float a1 = mat(0, 0) * mat(1, 2) - mat(0, 2) * mat(1, 0);
182 const float a2 = mat(0, 0) * mat(1, 3) - mat(0, 3) * mat(1, 0);
183 const float a3 = mat(0, 1) * mat(1, 2) - mat(0, 2) * mat(1, 1);
184 const float a4 = mat(0, 1) * mat(1, 3) - mat(0, 3) * mat(1, 1);
185 const float a5 = mat(0, 2) * mat(1, 3) - mat(0, 3) * mat(1, 2);
186 const float b0 = mat(2, 0) * mat(3, 1) - mat(2, 1) * mat(3, 0);
187 const float b1 = mat(2, 0) * mat(3, 2) - mat(2, 2) * mat(3, 0);
188 const float b2 = mat(2, 0) * mat(3, 3) - mat(2, 3) * mat(3, 0);
189 const float b3 = mat(2, 1) * mat(3, 2) - mat(2, 2) * mat(3, 1);
190 const float b4 = mat(2, 1) * mat(3, 3) - mat(2, 3) * mat(3, 1);
191 const float b5 = mat(2, 2) * mat(3, 3) - mat(2, 3) * mat(3, 2);
192 return (a0 * b5 - a1 * b4 + a2 * b3 + a3 * b2 - a4 * b1 + a5 * b0);
193 }

388 void decompose(Vector3T<T>* translation, QuaternionT<T>* rotation, Vector3T<T>* scale) const
389 {
390 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
391
392 auto& mat = *this;
393 if(translation)
394 {
395 translation->x = mat(3, 0);
396 translation->y = mat(3, 1);
397 translation->z = mat(3, 2);
398 }
399
400 if(scale == nullptr && rotation == nullptr)
401 return;
402
403 auto axisX = Vector3T<T>(mat(0, 0), mat(0, 1), mat(0, 2));
404 auto axisY = Vector3T<T>(mat(1, 0), mat(1, 1), mat(1, 2));
405 auto axisZ = Vector3T<T>(mat(2, 0), mat(2, 1), mat(2, 2));
406 auto scaleX = axisX.norm();
407 auto scaleY = axisY.norm();
408 auto scaleZ = axisZ.norm();
409 if(determinant() < 0)
410 scaleZ = -scaleZ;
411
412 if(scale)
413 {
414 scale->x = scaleX;
415 scale->y = scaleY;
416 scale->z = scaleZ;
417 }
418
419 if(rotation)
420 {
421 if(scaleX < std::numeric_limits<T>::epsilon() || scaleY < std::numeric_limits<T>::epsilon() ||
422 scaleZ < std::numeric_limits<T>::epsilon())
423 return; // 除数过于接近 0, 无法完成计算
424
425 axisX.normalize();
426 axisY.normalize();
427 axisZ.normalize();
428 const float trace = axisX.x + axisY.y + axisZ.z + 1.0f;
429
430 if(trace > std::numeric_limits<T>::epsilon())
431 {
432 const float s = 0.5f / std::sqrt(trace);
433 rotation->w = 0.25f / s;
434 rotation->x = (axisY.z - axisZ.y) * s;
435 rotation->y = (axisZ.x - axisX.z) * s;
436 rotation->z = (axisX.y - axisY.x) * s;
437 }
438 else
439 {
440 if(axisX.x > axisY.y && axisX.x > axisZ.z)
441 {
442 const float s = 0.5f / std::sqrt(1.0f + axisX.x - axisY.y - axisZ.z);
443 rotation->w = (axisY.z - axisZ.y) * s;
444 rotation->x = 0.25f / s;
445 rotation->y = (axisY.x + axisX.y) * s;
446 rotation->z = (axisZ.x + axisX.z) * s;
447 }
448 else if(axisY.y > axisZ.z)
449 {
450 const float s = 0.5f / std::sqrt(1.0f + axisY.y - axisX.x - axisZ.z);
451 rotation->w = (axisZ.x - axisX.z) * s;
452 rotation->x = (axisY.x + axisX.y) * s;
453 rotation->y = 0.25f / s;
454 rotation->z = (axisZ.y + axisY.z) * s;
455 }
456 else
457 {
458 const float s = 0.5f / std::sqrt(1.0f + axisZ.z - axisX.x - axisY.y);
459 rotation->w = (axisX.y - axisY.x) * s;
460 rotation->x = (axisZ.x + axisX.z) * s;
461 rotation->y = (axisZ.y + axisY.z) * s;
462 rotation->z = 0.25f / s;
463 }
464 }
465 }

749 static MatrixT createRotation(const QuaternionT<T>& q)
750 {
751 float x2 = q.x + q.x;
752 float y2 = q.y + q.y;
753 float z2 = q.z + q.z;
754
755 float xx2 = q.x * x2;
756 float yy2 = q.y * y2;
757 float zz2 = q.z * z2;
758 float xy2 = q.x * y2;
759 float xz2 = q.x * z2;
760 float yz2 = q.y * z2;
761 float wx2 = q.w * x2;
762 float wy2 = q.w * y2;
763 float wz2 = q.w * z2;
764
765 MatrixT mat;
766 mat(0, 0) = 1.0f - yy2 - zz2;
767 mat(0, 1) = xy2 + wz2;
768 mat(0, 2) = xz2 - wy2;
769
770 mat(1, 0) = xy2 - wz2;
771 mat(1, 1) = 1.0f - xx2 - zz2;
772 mat(1, 2) = yz2 + wx2;
773
774 mat(2, 0) = xz2 + wy2;
775 mat(2, 1) = yz2 - wx2;
776 mat(2, 2) = 1.0f - xx2 - yy2;
777 return mat;
778 }

786 static MatrixT createRotation(float angle, Vector3T<T> axis)
787 {
788 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
789 check(axis == Vector3T<T>::unit_x || axis == Vector3T<T>::unit_y || axis == Vector3T<T>::unit_z);
790
791 axis.normalize();
792 const float x = axis.x;
793 const float y = axis.y;
794 const float z = axis.z;
795
796 const float sin = std::sin(angle);
797 const float cos = std::cos(angle);
798 const float t = 1.0f - cos;
799 const float tx = t * x;
800 const float ty = t * y;
801 const float tz = t * z;
802 const float txy = tx * y;
803 const float txz = tx * z;
804 const float tyz = ty * z;
805 const float sx = sin * x;
806 const float sy = sin * y;
807 const float sz = sin * z;
808
809 auto mat = MatrixT::identity();
810 mat(0, 0) = cos + tx * x;
811 mat(0, 1) = txy + sz;
812 mat(0, 2) = txz - sy;
813
814 mat(1, 0) = txy - sz;
815 mat(1, 1) = cos + ty * y;
816 mat(1, 2) = tyz + sx;
817
818 mat(2, 0) = txz + sy;
819 mat(2, 1) = tyz - sx;
820 mat(2, 2) = cos + tz * z;
821 return mat;
822 }

975 static MatrixT<T, 4, 4> lookAt(const Vector3T<T>& eye, const Vector3T<T>& target,
976 const Vector3T<T>& up = Vector3T<T>::up)
977 {
978 static_assert(R == C && R == 4, "only 4x4 matrix supports this operation");
979 static_assert(std::is_floating_point_v<T>);
980
981 Vector3T<T> f((target - eye).normalize()); // front
982 Vector3T<T> u(up.normalized()); // up
983 Vector3T<T> r(f.cross(up).normalize()); // right
984 u = r.cross(f);
985
986 auto mat = MatrixT::identity();
987
988 mat(0, 0) = r.x;
989 mat(1, 0) = r.y;
990 mat(2, 0) = r.z;
991
992 mat(0, 1) = u.x;
993 mat(1, 1) = u.y;
994 mat(2, 1) = u.z;
995
996 mat(0, 2) = -f.x;
997 mat(1, 2) = -f.y;
998 mat(2, 2) = -f.z;
999
1000 mat(3, 0) = -r.dot(eye);
1001 mat(3, 1) = -u.dot(eye);
1002 mat(3, 2) = f.dot(eye);
1003
1004 return mat;
1005 }
