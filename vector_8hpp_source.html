250 friend std::ostream& operator<<(std::ostream& stream, const VectorT& vec)
251 {
252 size_t max = 0;
253 for(size_t i = 0; i < N; i++)
254 {
255 std::ostringstream stm;
256 stm << vec[i];
257 const auto size = stm.str().size();
258 max = std::max(max, size);
259 }
260 max = std::min<size_t>(max, 6);
261 for(size_t i = 0; i < N; i++)
262 {
263 std::ostringstream stm;
264 stm.setf(std::ios::fixed);
265 stm.precision(6);
266 stm << vec[i];
267 auto str = stm.str();
268 str.erase(str.find_last_not_of('0') + 1);
269 if(str.back() == '.')
270 str.pop_back();
271 stream.width(max);
272 stream << str << ' ';
273 }
274 return stream;
275 }
