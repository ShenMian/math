162 std::ostringstream stm;

164 const auto size = stm.str().size();
165 max = std::max(max, size);

167 max = std::min<size_t>(max, 6);

170 std::ostringstream stm;
171 stm.setf(std::ios::fixed);
172 stm.precision(6);
173 stm << vec[i];
174 auto str = stm.str();
175 str.erase(str.find_last_not_of('0') + 1);
176 if(str.back() == '.')
177 str.pop_back();
178 stream.width(max);
179 stream << str << ' ';
180 }
181 return stream;
