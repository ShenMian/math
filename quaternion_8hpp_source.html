44 constexpr explicit QuaternionT(const Matrix4T<T>& mat)
45 {
46 const auto trace = mat.trace();
47 if(trace > T(0))
48 {
49 const auto s = T(2) * detail::sqrt(trace);
50 x = (mat(2, 1) - mat(1, 2)) / s;
51 y = (mat(0, 2) - mat(2, 0)) / s;
52 z = (mat(1, 0) - mat(0, 1)) / s;
53 w = T(0.25) * s;
54 }
55 else if(mat(0, 0) > mat(1, 1) && mat(0, 0) > mat(2, 2))
56 {
57 const auto s = T(2) * detail::sqrt(T(1) + mat(0, 0) - mat(1, 1) - mat(2, 2));
58 x = T(0.25) * s;
59 y = (mat(0, 1) + mat(1, 0)) / s;
60 z = (mat(2, 0) + mat(0, 2)) / s;
61 w = (mat(2, 1) - mat(1, 2)) / s;
62 }
63 else if(mat(1, 1) > mat(2, 2))
64 {
65 const auto s = T(2) * detail::sqrt(T(1) + mat(1, 1) - mat(0, 0) - mat(2, 2));
66 x = (mat(0, 1) + mat(1, 0)) / s;
67 y = T(0.25) * s;
68 z = (mat(1, 2) + mat(2, 1)) / s;
69 w = (mat(0, 2) - mat(2, 0)) / s;
70 }
71 else
72 {
73 const auto s = T(2) * detail::sqrt(T(1) + mat(2, 2) - mat(0, 0) - mat(1, 1));
74 x = (mat(0, 2) + mat(2, 0)) / s;
75 y = (mat(1, 2) + mat(2, 1)) / s;
76 z = T(0.25) * s;
77 w = (mat(1, 0) - mat(0, 1)) / s;
78 }
79 }

86 constexpr void eular(const Vector3T<T>& angles) noexcept
87 {
88 const auto cr = detail::cos(angles.x / T(2));
89 const auto cp = detail::cos(angles.y / T(2));
90 const auto cy = detail::cos(angles.z / T(2));
91
92 const auto sr = detail::sin(angles.x / T(2));
93 const auto sp = detail::sin(angles.y / T(2));
94 const auto sy = detail::sin(angles.z / T(2));
95
96 const auto cpcy = cp * cy;
97 const auto spsy = sp * sy;
98 const auto cpsy = cp * sy;
99 const auto spcy = sp * cy;
100
101 x = sr * cpcy - cr * spsy;
102 y = cr * spcy + sr * cpsy;
103 z = cr * cpsy - sr * spcy;
104 w = cr * cpcy + sr * spsy;
105
106 normalize();
107 }

114 constexpr Vector3T<T> eular() const noexcept
115 {
116 Vector3T<T> angles;
117
118 const auto xx = x * x;
119 const auto yy = y * y;
120 const auto zz = z * z;
121 const auto ww = w * w;
122
123 angles.x = detail::atan2(T(2) * (y * z + x * w), -xx - yy + zz + ww);
124 angles.y = detail::asin(std::clamp(T(2) * (y * w - x * z), T(-1), T(1)));
125 angles.z = detail::atan2(T(2) * (x * y + z * w), xx - yy - zz + ww);
126
127 return angles;
128 }

133 constexpr QuaternionT& inverse()
134 {
135 T n = sizeSq();
136 if(n == T(1))
137 {
138 x = -x;
139 y = -y;
140 z = -z;
141 return *this;
142 }
143
144 if(n < std::numeric_limits<T>::epsilon())
145 return *this;
146
147 n = 1.0f / n;
148 x = -x * n;
149 y = -y * n;
150 z = -z * n;
151 w = w * n;
152 return *this;
153 }
