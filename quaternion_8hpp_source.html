43 explicit QuaternionT(const Matrix4T<T>& mat)
44 {
45 const auto trace = mat(0, 0) + mat(1, 1) + mat(2, 2) + T(1);
46 if(trace > T(0))
47 {
48 const auto s = T(2) * std::sqrt(trace);
49 x = (mat(2, 1) - mat(1, 2)) / s;
50 y = (mat(0, 2) - mat(2, 0)) / s;
51 z = (mat(1, 0) - mat(0, 1)) / s;
52 w = T(0.25) * s;
53 }
54 else if(mat(0, 0) > mat(1, 1) && mat(0, 0) > mat(2, 2))
55 {
56 const auto s = T(2) * std::sqrt(T(1) + mat(0, 0) - mat(1, 1) - mat(2, 2));
57 x = T(0.25) * s;
58 y = (mat(0, 1) + mat(1, 0)) / s;
59 z = (mat(2, 0) + mat(0, 2)) / s;
60 w = (mat(2, 1) - mat(1, 2)) / s;
61 }
62 else if(mat(1, 1) > mat(2, 2))
63 {
64 const auto s = T(2) * std::sqrt(T(1) + mat(1, 1) - mat(0, 0) - mat(2, 2));
65 x = (mat(0, 1) + mat(1, 0)) / s;
66 y = T(0.25) * s;
67 z = (mat(1, 2) + mat(2, 1)) / s;
68 w = (mat(0, 2) - mat(2, 0)) / s;
69 }
70 else
71 {
72 const auto s = T(2) * std::sqrt(T(1) + mat(2, 2) - mat(0, 0) - mat(1, 1));
73 x = (mat(0, 2) + mat(2, 0)) / s;
74 y = (mat(1, 2) + mat(2, 1)) / s;
75 z = T(0.25) * s;
76 w = (mat(1, 0) - mat(0, 1)) / s;
77 }
78 }

85 void eular(const Vector3T<T>& angles) noexcept
86 {
87 const auto cr = std::cos(angles.x / T(2));
88 const auto cp = std::cos(angles.y / T(2));
89 const auto cy = std::cos(angles.z / T(2));
90
91 const auto sr = std::sin(angles.x / T(2));
92 const auto sp = std::sin(angles.y / T(2));
93 const auto sy = std::sin(angles.z / T(2));
94
95 const auto cpcy = cp * cy;
96 const auto spsy = sp * sy;
97 const auto cpsy = cp * sy;
98 const auto spcy = sp * cy;
99
100 x = sr * cpcy - cr * spsy;
101 y = cr * spcy + sr * cpsy;
102 z = cr * cpsy - sr * spcy;
103 w = cr * cpcy + sr * spsy;
104
105 normalize();
106 }

113 Vector3T<T> eular() const noexcept
114 {
115 Vector3T<T> angles;
116
117 const auto xx = x * x;
118 const auto yy = y * y;
119 const auto zz = z * z;
120 const auto ww = w * w;
121
122 angles.x = std::atan2(T(2) * (y * z + x * w), -xx - yy + zz + ww);
123 angles.y = std::asin(std::clamp(T(2) * (y * w - x * z), T(-1), T(1)));
124 angles.z = std::atan2(T(2) * (x * y + z * w), xx - yy - zz + ww);
125
126 return angles;
127 }

132 QuaternionT& inverse()
133 {
134 const T n = sizeSq();
135 if(n == T(1))
136 {
137 x = -x;
138 y = -y;
139 z = -z;
140 return *this;
141 }
142
143 if(n < std::numeric_limits<T>::epsilon())
144 return *this;
145
146 n = 1.0f / n;
147 x = -x * n;
148 y = -y * n;
149 z = -z * n;
150 w = w * n;
151 return *this;
152 }
