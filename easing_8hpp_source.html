337 static float easeInElastic(float t, float b, float c, float d)
338 {
339 float s = 1.70158f;
340 float p = 0;
341 float a = c;
342 if(t == 0)
343 return b;
344 if((t /= d) == 1)
345 return b + c;
346 if(!p)
347 p = d * .3f;
348 if(a < abs(c))
349 {
350 a = c;
351 s = p / 4;
352 }
353 else
354 s = p / (2 * std::numbers::pi) * asin(c / a);
355 return -(a * pow(2, 10 * (--t)) * sin((t * d - s) * (2 * std::numbers::pi) / p)) + b;
356 }

463 {

466 else if(t < (2 / 2.75f))
467 return c * (7.5625f * (t -= (1.5f / 2.75f)) * t + .75f) + b;
468 else if(t < (2.5f / 2.75f))
469 return c * (7.5625f * (t -= (2.25f / 2.75f)) * t + .9375f) + b;
470 else
471 return c * (7.5625f * (t -= (2.625f / 2.75f)) * t + .984375f) + b;
472 }
